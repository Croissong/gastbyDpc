* Tabelle erstellen
#+BEGIN_SRC sql
CREATE TABLE IF NOT EXISTS zimmer (
        zimmer_id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
        bezeichnung VARCHAR(45),
        beschreibung VARCHAR(45));
#+END_SRC
